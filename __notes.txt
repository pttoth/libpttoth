bool Config::
        _isValidChar(char c) const{
    char* valids = "_-./";
    if( isalnum(c) ){ return true; } //A-Z, a-z, 0-9
    for(size_t i=0; i<strlen(valids); ++i){
        if( c == valids[i] ){ return true; }
    }
    return false;
}
